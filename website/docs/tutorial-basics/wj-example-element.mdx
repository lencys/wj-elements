import BrowserOnly from '@docusaurus/BrowserOnly';
import ExecutionEnvironment from '@docusaurus/ExecutionEnvironment';

import { useEffect } from 'react';

# Example element

```html title="Zápis v HTML"
<wj-example-element text="Hello word"></wj-example-element>
```

```js title="Zápis v Javascripte"
    let exampleElement = document.querySelector('wj-example-element');
    exampleElement.setAttribute('text', 'Hello word');
```

## Usage


export const Webcomponent = () => {
    useEffect(()=>{
        if (ExecutionEnvironment.canUseDOM){
            import("../../../dist/wj-main.js")
        }
    }, []);
    
    return (
        <BrowserOnly fallback={<div>Loading...</div>}>
            {() => {
                return <wj-example-element text="Hello word"></wj-example-element>;
            }}
        </BrowserOnly>
    )
};


<Webcomponent></Webcomponent>
